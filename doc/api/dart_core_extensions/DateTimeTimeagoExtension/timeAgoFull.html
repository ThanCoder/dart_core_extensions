String timeAgoFull() {
  final diff = DateTime.now().difference(this);

  if (diff.inSeconds < 5) return 'just now';
  if (diff.inMinutes < 1) {
    return '${diff.inSeconds} seconds ago';
  }
  if (diff.inHours < 1) {
    return '${diff.inMinutes} minutes ago';
  }
  if (diff.inDays < 1) {
    return '${diff.inHours} hours ago';
  }
  if (diff.inDays == 1) {
    return 'yesterday';
  }
  return '${diff.inDays} days ago';
}
